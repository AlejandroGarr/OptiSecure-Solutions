/**
 * @description Endpoint REST para recibir alertas de detecci├│n de personas
 *              desde el servidor Python (YOLOv8). Crea un registro Task y
 *              env├¡a un email de alerta al contacto asociado a la c├ímara.
 * @author      OptiSecure Solutions
 * @date        2026-04-05
 */
@RestResource(urlMapping='/PersonDetection/*')
global without sharing class PersonDetectionController {

    /**
     * @description Recibe una alerta de detecci├│n de persona.
     *              Body JSON esperado: { "cameraId": "...", "timestamp": "..." }
     * @return      Mensaje de confirmaci├│n o error.
     */
    @HttpPost
    global static String handleDetection() {
        RestRequest req = RestContext.request;
        Map<String, Object> body = (Map<String, Object>) JSON.deserializeUntyped(
            req.requestBody.toString()
        );

        String cameraId    = (String) body.get('cameraId');
        String timestampStr = (String) body.get('timestamp');
        String clipUrl      = (String) body.get('clipUrl');

        if (String.isBlank(cameraId)) {
            RestContext.response.statusCode = 400;
            return 'Error: falta el campo cameraId';
        }

        try {
            // Obtener datos de la c├ímara y su contacto asociado
            List<Camara__c> cams = [
                SELECT Id, Name, Zone__c,
                       Contacto__c,
                       Contacto__r.Name,
                       Contacto__r.Email
                FROM   Camara__c
                WHERE  Id = :cameraId
                LIMIT  1
            ];

            if (cams.isEmpty()) {
                RestContext.response.statusCode = 404;
                return 'Error: c├ímara no encontrada con Id ' + cameraId;
            }

            Camara__c cam = cams[0];

            // ÔöÇÔöÇÔöÇ Crear Task de actividad ÔöÇÔöÇÔöÇ
            Task t = new Task(
                Subject     = '\u26A0 Persona detectada - ' + cam.Name,
                Description = 'Se ha detectado una persona en la c├ímara "'
                            + cam.Name + '"'
                            + (cam.Zone__c != null ? ' (Zona: ' + cam.Zone__c + ')' : '')
                            + ' el ' + timestampStr + '.\n'
                            + 'Se ha grabado un clip de 8 segundos autom\u00e1ticamente.'
                            + (String.isNotBlank(clipUrl) ? '\n[CLIP_URL]' + clipUrl + '[/CLIP_URL]' : ''),
                Status       = 'Completed',
                Priority     = 'High',
                WhoId        = cam.Contacto__c,
                ActivityDate = Date.today()
            );
            insert t;

            // ÔöÇÔöÇÔöÇ Enviar email al contacto ÔöÇÔöÇÔöÇ
            if (cam.Contacto__c != null && cam.Contacto__r.Email != null) {
                sendAlertEmail(cam, timestampStr, clipUrl);
            }

            return 'OK - Alerta registrada para c├ímara ' + cam.Name;
        } catch (Exception e) {
            RestContext.response.statusCode = 500;
            return 'Error: ' + e.getMessage();
        }
    }

    /**
     * @description Env├¡a un email HTML formateado al contacto de la c├ímara.
     */
    private static void sendAlertEmail(Camara__c cam, String timestampStr, String clipUrl) {
        Messaging.SingleEmailMessage mail = new Messaging.SingleEmailMessage();
        mail.setToAddresses(new List<String>{ cam.Contacto__r.Email });
        mail.setSubject('Alerta de Seguridad OptiSecure - Persona Detectada');

        String clipSection = '';
        if (String.isNotBlank(clipUrl)) {
            clipSection = '<tr>'
                + '<td style="padding: 10px 14px; border: 1px solid #d1d5db; background: #f3f4f6; font-weight: bold;">Clip de v&iacute;deo</td>'
                + '<td style="padding: 10px 14px; border: 1px solid #d1d5db;">'
                + '<a href="' + clipUrl.escapeHtml4() + '" style="color: #2563eb; font-weight: bold; text-decoration: none;">&#x1F3AC; Ver clip de seguridad</a>'
                + '</td>'
                + '</tr>';
        }

        mail.setHtmlBody(
            '<div style="font-family: Arial, sans-serif; max-width: 600px; margin: 0 auto;">'
            + '<div style="background: #dc2626; color: white; padding: 20px; border-radius: 8px 8px 0 0;">'
            + '<h2 style="margin: 0;">&#x1F6A8; Alerta de Seguridad</h2>'
            + '</div>'
            + '<div style="background: #f9fafb; padding: 24px; border: 1px solid #e5e7eb; border-radius: 0 0 8px 8px;">'
            + '<p>Estimado/a <strong>' + cam.Contacto__r.Name + '</strong>,</p>'
            + '<p>Se ha detectado una <strong style="color: #dc2626;">persona</strong> en su c&aacute;mara de seguridad:</p>'
            + '<table style="border-collapse: collapse; width: 100%; margin: 16px 0;">'
            + '<tr>'
            + '<td style="padding: 10px 14px; border: 1px solid #d1d5db; background: #f3f4f6; font-weight: bold; width: 140px;">C&aacute;mara</td>'
            + '<td style="padding: 10px 14px; border: 1px solid #d1d5db;">' + cam.Name + '</td>'
            + '</tr>'
            + (cam.Zone__c != null
                ? '<tr>'
                  + '<td style="padding: 10px 14px; border: 1px solid #d1d5db; background: #f3f4f6; font-weight: bold;">Zona</td>'
                  + '<td style="padding: 10px 14px; border: 1px solid #d1d5db;">' + cam.Zone__c + '</td>'
                  + '</tr>'
                : '')
            + '<tr>'
            + '<td style="padding: 10px 14px; border: 1px solid #d1d5db; background: #f3f4f6; font-weight: bold;">Fecha / Hora</td>'
            + '<td style="padding: 10px 14px; border: 1px solid #d1d5db;">' + timestampStr + '</td>'
            + '</tr>'
            + clipSection
            + '</table>'
            + '<p>Se ha grabado un clip de <strong>8 segundos</strong> que puede revisar desde su panel de Salesforce, en la secci&oacute;n <em>Clips de Seguridad</em>.</p>'
            + '<hr style="border: none; border-top: 1px solid #e5e7eb; margin: 20px 0;">'
            + '<p style="color: #6b7280; font-size: 12px;">Este mensaje ha sido generado autom&aacute;ticamente por el sistema de vigilancia OptiSecure Solutions.</p>'
            + '</div>'
            + '</div>'
        );
        mail.setSaveAsActivity(false);

        Messaging.sendEmail(new List<Messaging.SingleEmailMessage>{ mail });
    }
}
